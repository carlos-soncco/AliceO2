37 bool DigitPixelReader::getNextChipData(ChipPixelData& chipData)
38 {
39  // decode data of single chip to chipData
40  if (!mLastDigit) {
41  if (mIdx >= mDigitArray->size()) {
42  return false;
43  }
44  mLastDigit = &((*mDigitArray)[mIdx++]);
45  }
46  chipData.clear();
47  chipData.setStartID(mIdx - 1);
48  chipData.setChipID(mLastDigit->getChipIndex());
49  chipData.setROFrame(mLastDigit->getROFrame());
50  chipData.getData().emplace_back(mLastDigit);
51  mLastDigit = nullptr;
52 
53  while (mIdx < mDigitArray->size()) {
54  mLastDigit = &((*mDigitArray)[mIdx++]);
55  if (chipData.getChipID() != mLastDigit->getChipIndex())
56  break;
57  if (chipData.getROFrame() != mLastDigit->getROFrame())
58  break;
59  chipData.getData().emplace_back(mLastDigit);
60  mLastDigit = nullptr;
61  }
62  return true;
63 }
