67  // In case no extra argument is provided and the passed type is trivially
68  // copyable and non polymorphic, the most likely wanted behavior is to create
69  // a message with that type, and so we do.
70  template <typename T>
71  typename std::enable_if<is_messageable<T>::value == true, T&>::type
72  make(const Output& spec)
73  {
74  DataChunk chunk = newChunk(spec, sizeof(T));
75  return *reinterpret_cast<T*>(chunk.data);
76  }

183  template <typename W>
184  typename std::enable_if<is_specialization<W, ROOTSerialized>::value == true, void>::type
185  snapshot(const Output& spec, W wrapper)
186  {
187  using T = typename W::wrapped_type;
188  static_assert(std::is_same<typename W::hint_type, const char>::value || //
189  std::is_same<typename W::hint_type, TClass>::value || //
190  std::is_void<typename W::hint_type>::value, //
191  "class hint must be of type TClass or const char");
192 
193  FairMQMessagePtr payloadMessage(mProxy.createMessage());
194  const TClass* cl = nullptr;
195  if (wrapper.getHint() == nullptr) {
196  // get TClass info by wrapped type
197  cl = TClass::GetClass(typeid(T));
198  } else if (std::is_same<typename W::hint_type, TClass>::value) {
199  // the class info has been passed directly
200  cl = reinterpret_cast<const TClass*>(wrapper.getHint());
201  } else if (std::is_same<typename W::hint_type, const char>::value) {
202  // get TClass info by optional name
203  cl = TClass::GetClass(reinterpret_cast<const char*>(wrapper.getHint()));
204  }
205  if (has_root_dictionary<T>::value == false && cl == nullptr) {
206  std::string msg("ROOT serialization not supported, dictionary not found for type ");
207  if (std::is_same<typename W::hint_type, const char>::value) {
208  msg += reinterpret_cast<const char*>(wrapper.getHint());
209  } else {
210  msg += typeid(T).name();
211  }
212  throw std::runtime_error(msg);
213  }
214  TMessageSerializer().Serialize(*payloadMessage, &wrapper(), cl);
215  addPartToContext(std::move(payloadMessage), spec, o2::header::gSerializationMethodROOT);
216  }

267  FairMQMessagePtr payloadMessage(mProxy.createMessage(sizeInBytes));

270  for (auto const & pointer : v) {
