40 InitStatus ClustererTask::Init()
41 {
43  FairRootManager* mgr = FairRootManager::Instance();
44  if (!mgr) {
45  LOG(ERROR) << "Could not instantiate FairRootManager. Exiting ..." << FairLogger::endl;
46  return kERROR;
47  }
48 
49  const auto arrDig = mgr->InitObjectAs<const std::vector<o2::ITSMFT::Digit>*>("ITSDigit");
50  if (!arrDig) {
51  LOG(ERROR) << "ITS digits are not registered in the FairRootManager. Exiting ..." << FairLogger::endl;
52  return kERROR;
53  }
54  mReader.setDigitArray(arrDig);
55 
56  const auto arrDigLbl = mUseMCTruth ? mgr->InitObjectAs<const MCTruth*>("ITSDigitMCTruth") : nullptr;
57 
58  if (!arrDigLbl && mUseMCTruth) {
59  LOG(WARNING) << "ITS digits labeals are not registered in the FairRootManager. Continue w/o MC truth ..."
60  << FairLogger::endl;
61  }
62  mClusterer.setDigitsMCTruthContainer(arrDigLbl);
63 
64  // Register output container
65  mgr->RegisterAny("ITSCluster", mClustersArrayPtr, kTRUE);
66 
67  // Register output MC Truth container if there is an MC truth for digits
68  if (arrDigLbl) {
69  mgr->RegisterAny("ITSClusterMCTruth", mClsLabelsPtr, kTRUE);
70  mClusterer.setClustersMCTruthContainer(mClsLabelsPtr);
71  }
72 
73  GeometryTGeo* geom = GeometryTGeo::Instance();
74  geom->fillMatrixCache(o2::utils::bit2Mask(o2::TransformType::T2L)); // make sure T2L matrices are loaded
75  mGeometry = geom;
76  mClusterer.setGeometry(geom);
77 
78  printf("%s | MCTruth: %s\n", Class()->GetName(), arrDigLbl ? "ON" : "OFF");
79  mClusterer.print();
80 
81  return kSUCCESS;
82 }
