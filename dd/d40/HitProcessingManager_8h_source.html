28 // a structure describing EventPart

30 struct EventPart {
31  EventPart() = default;
32  EventPart(int s, int e) : sourceID(s), entryID(e) {}
33  int sourceID = 0; // the ID of the source (0->backGround; > 1 signal source)

149 inline void HitProcessingManager::sampleCollisionConstituents()
150 {
151  auto getBackgroundRoundRobin = [this]() {
152  static int bgcounter = 0;
153  int numbg = mSimChains[0]->GetEntries();
154  if (bgcounter == numbg) {
155  bgcounter = 0;
156  }
157  return EventPart(0, bgcounter++);
158  };
159 
160  const int nsignalids = mSimChains.size() - 1;
161  auto getSignalRoundRobin = [this, nsignalids]() {
162  static int bgcounter = 0;
163  static int signalid = 0;
164  static std::vector<int> counter(nsignalids, 0);
165  if (signalid == nsignalids) {
166  signalid = 0;
167  }
168  const auto realsourceid = signalid + 1;
169  int numentries = mSimChains[realsourceid]->GetEntries();
170  if (counter[signalid] == numentries) {
171  counter[signalid] = 0;
172  }
173  EventPart e(realsourceid, counter[signalid]);
174  counter[signalid]++;
175  signalid++;
176  return e;
177  };
178 
179  // we fill mRunContext.mEventParts
180  auto& eventparts = mRunContext.mEventParts;
181  eventparts.clear();
182  eventparts.resize(mRunContext.mEventRecords.size());
183  for (int i = 0; i < mRunContext.mEventRecords.size(); ++i) {
184  eventparts[i].clear();
185  // push any number of constituents?
186  // for the moment just 2 : one background and one signal
187  eventparts[i].emplace_back(getBackgroundRoundRobin());
188  if (mSimChains.size() > 1) {
189  eventparts[i].emplace_back(getSignalRoundRobin());
190  }
191  }
192 
193  // push any number of constituents?
194  // for the moment just max 2 : one background and one signal
195  mRunContext.setMaxNumberParts(1);
196  if (mSimChains.size() > 1) {
197  mRunContext.setMaxNumberParts(2);
198  }
199 
200  mRunContext.printCollisionSummary();
201 }

212 template <typename HitType, typename Task_t>
213 std::function<void(const o2::steer::RunContext&)> defaultRunFunction(Task_t& task, std::string_view brname)
214 {
215  // using HitType = Task_t::InputType;
216  return [&task, brname](const o2::steer::RunContext& c) {
217  HitType* hittype = nullptr;
218  auto br = c.getBranch(brname.data());
219  assert(br);
220  br->SetAddress(&hittype);
221  for (auto entry = 0; entry < c.getNCollisions(); ++entry) {
222  br->GetEntry(entry);
223  task.setData(hittype, &c);
224  task.Exec("");
225  }
226  task.FinishTask();
227  // delete hittype
228  };
229 }
