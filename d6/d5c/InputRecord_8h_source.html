138  DataRef getByPos(int pos) const {
139  if (pos*2+1 > mCache.size() || pos < 0) {
140  throw std::runtime_error("Unknown argument requested at position " + std::to_string(pos));
141  }
142  if (pos > mInputsSchema.size()) {
143  throw std::runtime_error("Unknown schema at position");
144  }
145  if (mCache[pos*2] != nullptr && mCache[pos*2+1] != nullptr) {
146  return DataRef{&mInputsSchema[pos].matcher,
147  static_cast<char const*>(mCache[pos*2]->GetData()),
148  static_cast<char const*>(mCache[pos*2+1]->GetData())};
149  } else {
150  return DataRef{ &mInputsSchema[pos].matcher, nullptr, nullptr };
151  }
152  }
