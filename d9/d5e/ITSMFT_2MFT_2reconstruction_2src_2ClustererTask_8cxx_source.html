42 InitStatus ClustererTask::Init()
43 {
45  FairRootManager* mgr = FairRootManager::Instance();
46  if (!mgr) {
47  LOG(ERROR) << "Could not instantiate FairRootManager. Exiting ..." << FairLogger::endl;
48  return kERROR;
49  }
50 
51  const auto arrDig = mgr->InitObjectAs<const std::vector<o2::ITSMFT::Digit>*>("MFTDigit");
52  if (!arrDig) {
53  LOG(ERROR) << "MFT digits are not registered in the FairRootManager. Exiting ..." << FairLogger::endl;
54  return kERROR;
55  }
56  mReader.setDigitArray(arrDig);
57 
58  const auto arrDigLbl = mUseMCTruth ? mgr->InitObjectAs<const MCTruth*>("MFTDigitMCTruth") : nullptr;
59 
60  if (!arrDigLbl && mUseMCTruth) {
61  LOG(WARNING) << "MFT digits labeals are not registered in the FairRootManager. Continue w/o MC truth ..."
62  << FairLogger::endl;
63  }
64  mClusterer.setDigitsMCTruthContainer(arrDigLbl);
65 
66  // Register output container
67  mgr->RegisterAny("MFTCluster", mClustersArrayPtr, kTRUE);
68 
69  // Register output MC Truth container if there is an MC truth for digits
70  if (arrDigLbl) {
71  mgr->RegisterAny("MFTClusterMCTruth", mClsLabelsPtr, kTRUE);
72  mClusterer.setClustersMCTruthContainer(mClsLabelsPtr);
73  }
74 
75  GeometryTGeo* geom = GeometryTGeo::Instance();
76  geom->fillMatrixCache(o2::utils::bit2Mask(o2::TransformType::T2L)); // make sure T2L matrices are loaded
77  mGeometry = geom;
78  mClusterer.setGeometry(geom);
79 
80  printf("%s | MCTruth: %s\n", Class()->GetName(), arrDigLbl ? "ON" : "OFF");
81  mClusterer.print();
82 
83  return kSUCCESS;
84 }
