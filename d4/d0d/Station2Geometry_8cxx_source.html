591 void createStation2Geometry(TGeoVolume& topVolume)
592 {
594 
595  // create a quadrant
596  auto quadrant = createQuadrant();
597 
598  // rotation matrices to place the quadrants in the half-chambers
599  auto rot0 = new TGeoRotation();
600  auto rot1 = new TGeoRotation("reflXZ", 90., 180., 90., 90., 180., 0.);
601  auto rot2 = new TGeoRotation("reflXY", 90., 180., 90., 270., 0., 0.);
602  auto rot3 = new TGeoRotation("reflYZ", 90., 0., 90., -90., 180., 0.);
603  std::array<TGeoRotation*, 4> rot = { rot0, rot1, rot2, rot3 };
604 
605  int detElemId[4];
606  detElemId[0] = 1; // quadrant I
607  detElemId[1] = 0; // quadrant II
608  detElemId[2] = 3; // quadrant III
609  detElemId[3] = 2; // quadrant IV
610 
611  // Build the two chambers
612  float posz = 0.;
613 
614  for (int ich = 3; ich < 5; ich++) {
615 
616  // create two half-chambers (new compared to AliRoot !)
617  auto in = new TGeoVolumeAssembly(Form("SC0%dI", ich));
618  auto out = new TGeoVolumeAssembly(Form("SC0%dO", ich));
619 
620  // Place the quadrant in the half-chambers
621  for (int i = 0; i < 4; i++) {
622  posz = kQuadZpos * TMath::Power(-1, detElemId[i]);
623 
624  if (detElemId[i] == 1 || detElemId[i] == 2) {
625  in->AddNode(quadrant, detElemId[i] + ich * 100, new TGeoCombiTrans(0., 0., posz, rot[i]));
626  } else
627  out->AddNode(quadrant, detElemId[i] + ich * 100, new TGeoCombiTrans(0., 0., posz, rot[i]));
628  }
629 
630  // place the half-chambers in the top volume
631  topVolume.AddNode(in, 2 * (ich - 1), new TGeoTranslation(0., 0., kChamberZpos[ich - 3]));
632  topVolume.AddNode(out, 2 * ich - 1, new TGeoTranslation(0., 0., kChamberZpos[ich - 3]));
633 
634  } // end of the chamber loop
635 }
