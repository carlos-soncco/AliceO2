38 DataProcessorSpec getCATrackerSpec()
39 {
40  auto initFunction = [](InitContext& ic) {
41  auto options = ic.options().get<std::string>("tracker-options");
42 
43  using ClusterGroupParser = o2::algorithm::ForwardParser<o2::TPC::ClusterGroupHeader>;
44  auto parser = std::make_shared<ClusterGroupParser>();
45  auto tracker = std::make_shared<o2::TPC::TPCCATracking>();
46  tracker->initialize(options.c_str());
47 
48  auto processingFct = [parser, tracker](ProcessingContext& pc) {
49  ClusterNativeAccessFullTPC clusterIndex;
50  memset(&clusterIndex, 0, sizeof(clusterIndex));
51 
52  for (const auto& ref : pc.inputs()) {
53  auto size = o2::framework::DataRefUtils::getPayloadSize(ref);
54  LOG(INFO) << " " << *(ref.spec) << ", size " << size;
55  parser->parse(ref.payload, size,
56  [](const typename ClusterGroupParser::HeaderType& h) {
57  // check the header, but in this case there is no validity check
58  return true;
59  },
60  [](const typename ClusterGroupParser::HeaderType& h) {
61  // get the size of the frame including payload
62  // and header and trailer size, e.g. payload size
63  // from a header member
64  return h.nClusters * sizeof(ClusterNative) + ClusterGroupParser::totalOffset;
65  },
66  [&](typename ClusterGroupParser::FrameInfo& frame) {
67  int sector = frame.header->sector;
68  int padrow = frame.header->globalPadRow;
69  int nClusters = frame.header->nClusters;
70  LOG(DEBUG) << " sector " << std::setw(2) << std::setfill('0') << sector << " padrow "
71  << std::setw(3) << std::setfill('0') << padrow << " clusters " << std::setw(3)
72  << std::setfill(' ') << nClusters;
73  clusterIndex.clusters[sector][padrow] = reinterpret_cast<const ClusterNative*>(frame.payload);
74  clusterIndex.nClusters[sector][padrow] = nClusters;
75 
76  return true;
77  });
78  }
79 
80  std::vector<TrackTPC> tracks;
81  int retVal = tracker->runTracking(clusterIndex, &tracks, nullptr);
82  if (retVal != 0) {
83  // FIXME: error policy
84  LOG(ERROR) << "tracker returned error code " << retVal;
85  }
86  LOG(INFO) << "found " << tracks.size() << " track(s)";
87  pc.outputs().snapshot(OutputRef{ "output" }, tracks);
88  };
89 
90  return processingFct;
91  };
92 
93  return DataProcessorSpec{ "tracker", // process id
94  { InputSpec{ "input", "TPC", "CLUSTERNATIVE", 0, Lifetime::Timeframe } },
95  { OutputSpec{ { "output" }, gDataOriginTPC, "TRACKS", 0, Lifetime::Timeframe } },
96  AlgorithmSpec(initFunction) };
97 }
