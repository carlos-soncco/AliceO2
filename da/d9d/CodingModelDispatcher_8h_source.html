92  template <typename T>
93  return_type operator()(T& stage)
94  {
95  // the addWeight function belongs to the probability model as base
96  // of the specific model; funcions of the base can be accessed by
97  // static casting. This avoids an extra level of function calls.
98  return static_cast<typename T::wrapped_type::base_type&>(*stage).addWeight(value, weight);
99  }

112  template <typename ValueType, typename WeightType>
113  bool addWeight(ValueType v, WeightType w, bool switchToNextModel = true)
114  {
115  bool result = mContainer.apply(mPosition, addWeightFctr<ValueType, WeightType>(v, w));
116  if (switchToNextModel && ++mPosition >= getNumberOfModels()) {

327  class readFctr
328  {
329  public:
330  readFctr(std::istream& in, container_type& container) : mIn(in), mContainer(container) {}
331  ~readFctr() {}
332 
333  using return_type = bool;
334 
335  template <typename T>
336  return_type operator()(boost::type<T>)
337  {
338  T& stage = static_cast<T&>(mContainer);
339  std::string level, name, remaining;
340  mIn >> level;
341  mIn >> name;
342  if (!mIn) {
343  return false;
344  }
345  if (std::stoi(level) != T::level::value || name.compare((*stage).getName())) {
346  std::cerr << "Format error: expecting level '" << T::level::value << "' and name '" << (*stage).getName()
347  << "', got: " << level << " " << name << std::endl;
348  }
349  std::cout << "reading configuration for model " << name << std::endl;
350  std::getline(mIn, remaining); // flush the current line
351  (*stage).read(mIn);
352  return true;
353  }
354 
355  private:
356  std::istream& mIn;
357  container_type& mContainer;
358  };
