52  for (; clusterIndex < nClusters; clusterIndex++) {
53  // clusters are supposed to be sorted in CRU number, start a new entry
54  // for every new CRU, a map is needed if the clusters are not ordered
55  auto inputcluster = inClusters[clusterIndex];
56  uint16_t CRU = inputcluster.getCRU();
57  if (containerMetrics.size() == 0 || containerMetrics.back().CRU != CRU) {
58  containerMetrics.emplace_back(ClusterHardwareContainer{ 0, 0, 0, 0, 0, 0, 0, 0, 0xFFFFFFFF, 0, CRU });
59  }
60  containerMetrics.back().numberOfClusters++;
61  if (inputcluster.getTimeMean() < containerMetrics.back().timeBinOffset) {
62  containerMetrics.back().timeBinOffset = inputcluster.getTimeMean();
63  }
64  }

66  ClusterHardwareContainer8kb clusterContainerMemory;
67  auto maxClustersPerContainer = clusterContainerMemory.getMaxNumberOfClusters();
68  unsigned nPages = 0;
69  // at this point there is at least one cluster per record
70  for (const auto& m : containerMetrics) {
71  LOG(DEBUG) << "CRU " << m.CRU << " " << m.numberOfClusters << " cluster(s)";
72  nPages += (m.numberOfClusters - 1) / maxClustersPerContainer + 1;
73  }
74  LOG(DEBUG) << "allocating " << nPages << " output page(s), " << nPages * sizeof(ClusterHardwareContainer8kb);
75  auto outputPages = pc.outputs().make<ClusterHardwareContainer8kb>(OutputRef{ "clusterout" }, nPages);
76 
77  auto containerMetricsIterator = containerMetrics.begin();
78  auto outputPageIterator = outputPages.begin();
79  clusterIndex = 0;
80  while (clusterIndex < nClusters) {
81  LOG(DEBUG) << "processing CRU " << containerMetricsIterator->CRU;
82  while (containerMetricsIterator->numberOfClusters > 0) {
83  ClusterHardwareContainer* clusterContainer = outputPageIterator->getContainer();
84  *clusterContainer = *containerMetricsIterator;
85  if (clusterContainer->numberOfClusters > maxClustersPerContainer) {
86  clusterContainer->numberOfClusters = maxClustersPerContainer;
87  }

93  const auto& inputCluster = inClusters[clusterIndex + clusterInPage];
