218 void displayHistory(const DeviceInfo& info, DeviceControl& control)
219 {
220  if (info.history.empty()) {
221  return;
222  }
223  int startPos = info.historyPos;
224  const int historySize = info.history.size();
225 
226  int triggerStartPos = startPos + 1 % historySize;
227  int triggerStopPos = startPos % historySize;
228 
229  int j = startPos;
230  // We look for a stop trigger, so that we know where to stop the search for
231  // out start search. If no stop trigger is found, we search until the end
232  if (control.logStopTrigger[0]) {
233  while ((j % historySize) != ((startPos + 1) % historySize)) {
234  assert(j >= 0);
235  assert(j < historySize);
236  auto& line = info.history[j];
237  if (strstr(line.c_str(), control.logStopTrigger)) {
238  triggerStopPos = (j + 1) % historySize;
239  break;
240  }
241  // Wrap in case we end up below 0
242  j = (j == 0) ? historySize - 1 : j - 1;
243  }
244  }
245 
246  // Look for the last instance of the start trigger before the
247  // last stop trigger.
248  j = startPos + 1;
249  if (control.logStartTrigger[0]) {
250  while ((j % historySize) != triggerStopPos) {
251  assert(historySize > j);
252  assert(historySize == 1000);
253  auto& line = info.history[j];
254  if (strstr(line.c_str(), control.logStartTrigger)) {
255  triggerStartPos = j;
256  }
257  j = (j + 1) % historySize;
258  }
259  }
260 
261  // We start from the last trigger found. Eventually this is the first
262  // line in the ring buffer, if no trigger is specified.
263  size_t ji = triggerStartPos % historySize;
264  size_t je = triggerStopPos % historySize;
265  size_t iterations = 0;
266  while (historySize && ((ji % historySize) != je)) {
267  assert(iterations < historySize);
268  iterations++;
269  assert(historySize == 1000);
270  assert(ji < historySize);
271  auto& line = info.history[ji];
272  auto logLevel = info.historyLevel[ji];
273 
274  // Skip empty lines
275  if (line.empty()) {
276  ji = (ji + 1) % historySize;
277  continue;
278  }
279  // Print matching lines
280  if (strstr(line.c_str(), control.logFilter) != 0) {
281  auto color = colorForLogLevel(logLevel);
282  // We filter twice, once on input, to reduce the
283  // stream, a second time at display time, to avoid
284  // showing unrelevant messages from past.
285  if (logLevel >= control.logLevel) {
286  ImGui::TextColored(color, line.c_str(), line.c_str() + line.size());
287  }
288  }
289  ji = (ji + 1) % historySize;
290  }
291 }

301 void historyBar(DeviceGUIState& state, const DeviceSpec& spec, const DeviceMetricsInfo& metricsInfo)
302 {
303  bool open = ImGui::TreeNode(state.label.c_str());
304  if (open) {
305  ImGui::Text("# channels: %lu", spec.outputChannels.size() + spec.inputChannels.size());
306  ImGui::TreePop();
307  }
308  ImGui::NextColumn();
309 
310  if (gState.selectedMetric == -1) {
311  ImGui::NextColumn();
312  return;
313  }
314 
315  auto currentMetricName = gState.availableMetrics[gState.selectedMetric];
316 
317  size_t i = DeviceMetricsHelper::metricIdxByName(currentMetricName, metricsInfo);
318  // We did not find any plot, skipping this.
319  if (i == metricsInfo.metricLabelsIdx.size()) {
320  ImGui::NextColumn();
321  return;
322  }
323  auto& metric = metricsInfo.metrics[i];
324 
325  switch (metric.type) {
326  case MetricType::Int: {
327  HistoData<int> data;
328  data.mod = metricsInfo.timestamps[i].size();
329  data.first = metric.pos - data.mod;
330  data.size = metricsInfo.intMetrics[metric.storeIdx].size();
331  data.points = metricsInfo.intMetrics[metric.storeIdx].data();
332 
333  auto getter = [](void* hData, int idx) -> float {
334  auto histoData = reinterpret_cast<HistoData<int>*>(hData);
335  size_t pos = (histoData->first + static_cast<size_t>(idx)) % histoData->mod;
336  assert(pos >= 0 && pos < 1024);
337  return histoData->points[pos];
338  };
339  ImGui::PlotLines(currentMetricName.c_str(), getter, &data, data.size);
340  ImGui::NextColumn();
341  } break;
342  case MetricType::Float: {
343  HistoData<float> data;
344  data.mod = metricsInfo.timestamps[i].size();
345  data.first = metric.pos - data.mod;
346  data.size = metricsInfo.floatMetrics[metric.storeIdx].size();
347  data.points = metricsInfo.floatMetrics[metric.storeIdx].data();
348 
349  auto getter = [](void* hData, int idx) -> float {
350  auto histoData = reinterpret_cast<HistoData<float>*>(hData);
351  size_t pos = (histoData->first + static_cast<size_t>(idx)) % histoData->mod;
352  assert(pos >= 0 && pos < 1024);
353  return histoData->points[pos];
354  };
355  ImGui::PlotLines(currentMetricName.c_str(), getter, &data, data.size);
356  ImGui::NextColumn();
357  } break;
358  default:
359  ImGui::NextColumn();
360  return;
361  break;
362  }

365 void displayDeviceHistograms(const std::vector<DeviceInfo>& infos, const std::vector<DeviceSpec>& devices,
366  std::vector<DeviceControl>& controls, const std::vector<DeviceMetricsInfo>& metricsInfos)
367 {
368  bool graphNodes = true;
369  showTopologyNodeGraph(&graphNodes, infos, devices);
370  ImGui::SetNextWindowPos(ImVec2(0, ImGui::GetIO().DisplaySize.y - 300), 0);
371  ImGui::SetNextWindowSize(ImVec2(ImGui::GetIO().DisplaySize.x, 300), 0);
372 
373  // Calculate the unique set of metrics, as available in the metrics service
374  std::set<std::string> allMetricsNames;
375  for (const auto& metricsInfo : metricsInfos) {
376  for (const auto& labelsPairs : metricsInfo.metricLabelsIdx) {
377  allMetricsNames.insert(labelsPairs.first);
378  }
379  }
380  using NamesIndex = std::vector<std::string>;
381  gState.availableMetrics.clear();
382  std::copy(allMetricsNames.begin(), allMetricsNames.end(), std::back_inserter(gState.availableMetrics));
383 
384  ImGui::Begin("Devices");
385  ImGui::Combo("Select metric", &gState.selectedMetric,
386  [](void* data, int idx, const char** outText) -> bool {
387  NamesIndex* v = reinterpret_cast<NamesIndex*>(data);
388  if (idx >= v->size()) {
389  return false;
390  }
391  *outText = v->at(idx).c_str();
392  return true;
393  },
394  &gState.availableMetrics, gState.availableMetrics.size());
395 
396  // Calculate the full timestamp range for the selected metric
397  if (gState.selectedMetric >= 0) {
398  auto currentMetricName = gState.availableMetrics[gState.selectedMetric];
399  size_t minTime = -1;
400  size_t maxTime = 0;
401  for (auto& metricInfo : metricsInfos) {
402  size_t mi = DeviceMetricsHelper::metricIdxByName(currentMetricName, metricInfo);
403  if (mi == metricInfo.metricLabelsIdx.size()) {
404  continue;
405  }
406  auto& metric = metricInfo.metrics[mi];
407  size_t minRangePos = metric.pos % metricInfo.timestamps.size();
408  size_t maxRangePos = (size_t)(metric.pos) - 1 % metricInfo.timestamps.size();
409  auto& timestamps = metricInfo.timestamps[mi];
410  size_t curMinTime = timestamps[minRangePos];
411  size_t curMaxTime = timestamps[maxRangePos];
412  minTime = minTime < curMinTime ? minTime : curMinTime;
413  maxTime = maxTime > curMaxTime ? maxTime : curMaxTime;
414  }
415  ImGui::Text("min timestamp: %zu, max timestamp: %zu", minTime, maxTime);
416  }
417  ImGui::BeginChild("ScrollingRegion", ImVec2(0, -ImGui::GetItemsLineHeightWithSpacing()), false,
418  ImGuiWindowFlags_HorizontalScrollbar);
419  ImGui::Columns(2);
420  ImGui::SetColumnOffset(1, 300);
421  for (size_t i = 0; i < gState.devices.size(); ++i) {
422  DeviceGUIState& guiState = gState.devices[i];
423  const DeviceSpec& spec = devices[i];
424  const DeviceMetricsInfo& metricsInfo = metricsInfos[i];
425 
426  historyBar(guiState, spec, metricsInfo);
427  }
428  ImGui::Columns(1);
429  ImGui::EndChild();
430  ImGui::End();
431 }
