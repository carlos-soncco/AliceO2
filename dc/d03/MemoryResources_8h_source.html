116 // This in general (as in STL) is a bad idea, but here it is safe to inherit from an allocator since we
117 // have no additional data and only override some methods so we don't get into slicing and other problems.
118 template <typename T>
119 class SpectatorAllocator : public boost::container::pmr::polymorphic_allocator<T>
120 {
121  public:
122  using boost::container::pmr::polymorphic_allocator<T>::polymorphic_allocator;
123 
124  // skip default construction of empty elements
125  // this is important for two reasons: one: it allows us to adopt an existing buffer (e.g. incoming message) and
126  // quickly construct large vectors while skipping the element initialization.
127  template <typename... U>
128  void construct(U*...){};
129 
130  // dont try to call destructors, makes no sense since resource is managed externally AND allowed
131  // types cannot have side effects
132  template <typename... U>
133  void destroy(U*...){};
134 };

186 // return the message associated with the container or nullptr if it does not make sense (e.g. when we are just
187 // watching an existing message or when the container is not using FairMQMemoryResource as backend).
188 template <typename ContainerT>
189 typename std::enable_if<std::is_base_of<boost::container::pmr::polymorphic_allocator<typename ContainerT::value_type>,
190  typename ContainerT::allocator_type>::value == true,
191  FairMQMessagePtr>::type
192  getMessage(ContainerT&& container_, o2::memoryResources::FairMQMemoryResource* targetResource = nullptr)
193 {
194  auto container = std::move(container_);
195  auto alloc = container.get_allocator();
196 
197  auto resource = dynamic_cast<o2::memoryResources::FairMQMemoryResource*>(alloc.resource());
198  if (!resource && !targetResource) {
199  throw std::runtime_error("Neither the container or target resource specified");
200  }
201  size_t containerSizeBytes = container.size() * sizeof(typename ContainerT::value_type);
202  if ((!targetResource && resource) || (resource && targetResource && resource->is_equal(*targetResource))) {
203  auto message = resource->getMessage(static_cast<void*>(container.data()));
204  message->SetUsedSize(containerSizeBytes);
205  return std::move(message);
206  } else {
207  auto message = targetResource->getTransportFactory()->CreateMessage(containerSizeBytes);
208  std::memcpy(static_cast<byte*>(message->GetData()), container.data(), containerSizeBytes);
209  return std::move(message);
210  }
211 };

214 // This returns a unique_ptr of const vector, does not allow modifications at the cost of pointer
215 // semantics for access.
216 // use auto or decltype to catch the return value (or use span)
217 template <typename ElemT>
218 auto adoptVector(size_t nelem, FairMQMessage* message)
219 {
220  using namespace o2::memoryResources;
221  using DataType = std::vector<ElemT, ByteSpectatorAllocator>;
222 
223  struct doubleDeleter {
224  // kids: don't do this at home! (but here it's OK)
225  // this stateful deleter allows a single unique_ptr to manage 2 resources at the same time.
226  std::unique_ptr<SpectatorMessageResource> extra;
227  void operator()(const DataType* ptr) { delete ptr; }
228  };
229 
230  using OutputType = std::unique_ptr<const DataType, doubleDeleter>;
231 
232  auto resource = std::make_unique<SpectatorMessageResource>(message);
233  auto output = new DataType(nelem, ByteSpectatorAllocator{ resource.get() });
234  return OutputType(output, doubleDeleter{ std::move(resource) });
